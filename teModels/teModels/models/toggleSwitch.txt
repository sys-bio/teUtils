       P1 -> ; k1*P1;

      $x -> P2; A2/(1 + P1^n); 
       P2 -> ; k2*P2;
       
      A1 = 1; A2 = 0.5; n = 4; k1 = 0.2; k2 = 0.075;
      P1 = 0; P2 = 5
''')
